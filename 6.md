# JavaScript Basics

JavaScript is a scripting language that runs in the browser (and on servers via Node.js).
It makes web pages interactive — handling clicks, updating content, validating forms, etc.

---

## 1. Where to Write JavaScript

### Inline (inside HTML)
```html
<button onclick="alert('Hello!')">Click Me</button>
```

### Internal (inside `<script>` tag)
```html
<script>
  console.log("Hello from script tag");
</script>
```

### External (separate `.js` file — recommended)
```html
<script src="app.js"></script>
```

## 2. Output / Debugging

```js
console.log("Hello, World!");   // prints to browser console
alert("Hello!");                // shows a popup
document.write("Hi");          // writes directly to the page (rarely used)
```

## 3. Variables

Variables store data. Use `let` or `const` (avoid old `var`).

```js
let name = "Alice";       // can be changed later
const age = 25;           // cannot be reassigned
let score = 0;

name = "Bob";             // ok
// age = 30;              // ERROR — const cannot be reassigned
```

| Keyword | Reassignable | Scope   |
|---------|-------------|---------|
| `var`   | Yes         | Function|
| `let`   | Yes         | Block   |
| `const` | No          | Block   |

## 4. Data Types

```js
let text    = "Hello";          // String
let number  = 42;               // Number
let decimal = 3.14;             // Number (no separate float type)
let isTrue  = true;             // Boolean
let nothing = null;             // Null (intentionally empty)
let unknown;                    // Undefined (not assigned)
let obj     = { key: "value" }; // Object
let arr     = [1, 2, 3];        // Array (special object)
```

Check the type with `typeof`:
```js
console.log(typeof "hello"); // "string"
console.log(typeof 42);      // "number"
console.log(typeof true);    // "boolean"
```

## 5. Operators

### Arithmetic
```js
let a = 10, b = 3;
console.log(a + b);  // 13
console.log(a - b);  // 7
console.log(a * b);  // 30
console.log(a / b);  // 3.333...
console.log(a % b);  // 1  (remainder)
console.log(a ** b); // 1000 (exponent)
```

### Comparison
```js
console.log(5 == "5");   // true  (loose — only checks value)
console.log(5 === "5");  // false (strict — checks value AND type)
console.log(5 !== 3);    // true
console.log(10 > 3);     // true
console.log(10 <= 10);   // true
```

> Always prefer `===` over `==` to avoid unexpected type coercion.

### Logical
```js
console.log(true && false); // false (AND)
console.log(true || false); // true  (OR)
console.log(!true);         // false (NOT)
```

## 6. Strings

```js
let first = "John";
let last  = "Doe";

// Concatenation
console.log(first + " " + last);       // "John Doe"

// Template Literals (backtick) — preferred
console.log(`Hello, ${first} ${last}!`); // "Hello, John Doe!"

// Useful string methods
let str = "  Hello World  ";
console.log(str.length);          // 15
console.log(str.trim());          // "Hello World"
console.log(str.toUpperCase());   // "  HELLO WORLD  "
console.log(str.toLowerCase());   // "  hello world  "
console.log(str.includes("World")); // true
console.log(str.replace("World", "JS")); // "  Hello JS  "
console.log("a,b,c".split(","));  // ["a", "b", "c"]
```

## 7. Conditionals

### if / else if / else
```js
let score = 75;

if (score >= 90) {
  console.log("Grade: A");
} else if (score >= 75) {
  console.log("Grade: B");
} else if (score >= 60) {
  console.log("Grade: C");
} else {
  console.log("Grade: F");
}
```

### Ternary Operator (short if/else)
```js
let age = 20;
let status = age >= 18 ? "Adult" : "Minor";
console.log(status); // "Adult"
```

### switch
```js
let day = "Monday";

switch (day) {
  case "Monday":
    console.log("Start of work week");
    break;
  case "Friday":
    console.log("End of work week");
    break;
  default:
    console.log("Midweek");
}
```

## 8. Loops

### for loop
```js
for (let i = 0; i < 5; i++) {
  console.log(i); // 0, 1, 2, 3, 4
}
```

### while loop
```js
let count = 0;
while (count < 3) {
  console.log(count); // 0, 1, 2
  count++;
}
```

### for...of (iterating arrays)
```js
let fruits = ["Apple", "Banana", "Cherry"];
for (let fruit of fruits) {
  console.log(fruit);
}
```

### for...in (iterating object keys)
```js
let person = { name: "Alice", age: 25, city: "NYC" };
for (let key in person) {
  console.log(key, ":", person[key]);
}
```

## 9. Functions

### Function Declaration
```js
function greet(name) {
  return "Hello, " + name + "!";
}
console.log(greet("Alice")); // "Hello, Alice!"
```

### Function Expression
```js
const add = function(a, b) {
  return a + b;
};
console.log(add(3, 4)); // 7
```

### Arrow Function (modern, shorter syntax)
```js
const multiply = (a, b) => a * b;
console.log(multiply(3, 4)); // 12

// Multi-line arrow function
const greetUser = (name) => {
  let message = `Welcome, ${name}!`;
  return message;
};
```

### Default Parameters
```js
function greet(name = "Guest") {
  return `Hello, ${name}!`;
}
console.log(greet());        // "Hello, Guest!"
console.log(greet("Alice")); // "Hello, Alice!"
```

## 10. Arrays

```js
let colors = ["red", "green", "blue"];

// Access
console.log(colors[0]);        // "red"
console.log(colors.length);    // 3

// Add / Remove
colors.push("yellow");         // add to end
colors.pop();                  // remove from end
colors.unshift("purple");      // add to beginning
colors.shift();                // remove from beginning

// Common methods
let nums = [3, 1, 4, 1, 5, 9];
console.log(nums.sort());      // [1, 1, 3, 4, 5, 9]
console.log(nums.reverse());   // [9, 5, 4, 3, 1, 1]
console.log(nums.includes(4)); // true
console.log(nums.indexOf(5));  // index of value 5

// map — transform each element
let doubled = [1, 2, 3].map(n => n * 2);
console.log(doubled); // [2, 4, 6]

// filter — keep elements that pass a test
let evens = [1, 2, 3, 4, 5].filter(n => n % 2 === 0);
console.log(evens); // [2, 4]

// reduce — accumulate to single value
let sum = [1, 2, 3, 4].reduce((acc, n) => acc + n, 0);
console.log(sum); // 10
```

## 11. Objects

```js
let person = {
  name: "Alice",
  age: 25,
  city: "New York",
  greet: function() {
    return `Hi, I'm ${this.name}`;
  }
};

// Access
console.log(person.name);       // dot notation
console.log(person["age"]);     // bracket notation

// Modify
person.age = 26;
person.email = "alice@mail.com"; // add new property

// Method call
console.log(person.greet());    // "Hi, I'm Alice"

// Destructuring
const { name, city } = person;
console.log(name, city); // "Alice" "New York"
```

## 12. DOM Manipulation

The **DOM (Document Object Model)** lets JavaScript interact with HTML elements.

```html
<p id="message">Hello</p>
<button id="btn">Change Text</button>

<script>
  // Select elements
  const msg = document.getElementById("message");
  const btn = document.getElementById("btn");

  // Change content
  btn.addEventListener("click", function() {
    msg.textContent = "Text changed!";
    msg.style.color = "red";
  });
</script>
```

### Common DOM Methods
```js
document.getElementById("id")           // by ID
document.querySelector(".class")        // first match (CSS selector)
document.querySelectorAll("p")          // all matches (NodeList)

element.textContent = "New text";       // change text
element.innerHTML  = "<b>Bold</b>";     // change HTML
element.style.color = "blue";           // change style
element.classList.add("active");        // add CSS class
element.classList.remove("active");     // remove CSS class
element.classList.toggle("active");     // toggle CSS class
```

## 13. Events

```js
const btn = document.querySelector("#btn");

// Click event
btn.addEventListener("click", () => {
  console.log("Button clicked!");
});

// Common events:
// "click"      — mouse click
// "mouseover"  — mouse enters element
// "mouseout"   — mouse leaves element
// "keydown"    — key pressed
// "submit"     — form submitted
// "input"      — input field value changed
// "load"       — page fully loaded
```

## 14. Error Handling

```js
try {
  let result = 10 / 0;        // Infinity, not an error in JS
  JSON.parse("not valid json"); // throws SyntaxError
} catch (error) {
  console.log("Error caught:", error.message);
} finally {
  console.log("This always runs");
}
```

## 15. ES6+ Modern Features (Quick Reference)

```js
// Spread operator
let a = [1, 2, 3];
let b = [...a, 4, 5];   // [1, 2, 3, 4, 5]

// Rest parameters
function sum(...nums) {
  return nums.reduce((acc, n) => acc + n, 0);
}
console.log(sum(1, 2, 3, 4)); // 10

// Short-circuit evaluation
let user = null;
let name = user?.name ?? "Guest"; // Optional chaining + nullish coalescing
console.log(name); // "Guest"
```
